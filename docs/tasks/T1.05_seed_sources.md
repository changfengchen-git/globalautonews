### scripts/seed_sources.py

```python
"""
站点导入脚本

用法：
    python scripts/seed_sources.py

功能：
    1. 读取 config/sources.yaml
    2. 对每个站点：
       - 如果 URL 不存在于数据库 → INSERT
       - 如果 URL 已存在 → UPDATE（更新 name, country, language, priority 等元数据，保留运行时学习到的字段）
    3. 输出导入统计

注意事项：
    - next_crawl_at 对新导入的站点设为 NOW()（立即开始抓取）
    - status 对新导入的站点设为 'active'
    - 已存在站点的 status, consecutive_errors, discovery_rate 等运行时字段不覆盖
"""
import asyncio
import sys
import os
from datetime import datetime, timezone
from pathlib import Path

import yaml

# 添加项目根目录到 path
sys.path.insert(0, str(Path(__file__).parent.parent))

from shared.database import get_engine, get_session
from shared.models import Source
from sqlalchemy import select


async def seed_sources():
    # 读取 YAML
    config_path = Path(__file__).parent.parent / "config" / "sources.yaml"
    with open(config_path, "r", encoding="utf-8") as f:
        config = yaml.safe_load(f)

    sources_data = config.get("sources", [])
    if not sources_data:
        print("❌ No sources found in config/sources.yaml")
        return

    engine = get_engine()
    inserted = 0
    updated = 0
    skipped = 0

    async with get_session(engine) as session:
        for item in sources_data:
            url = item["url"].strip()
            # 检查是否已存在
            result = await session.execute(
                select(Source).where(Source.url == url)
            )
            existing = result.scalar_one_or_none()

            if existing:
                # 更新元数据，保留运行时字段
                existing.name = item["name"]
                existing.domain = item["domain"]
                existing.country = item["country"]
                existing.language = item["language"]
                existing.region = item.get("region")
                existing.priority = item.get("priority", "medium")
                existing.tier = item.get("tier", 1)
                updated += 1
            else:
                # 新增
                source = Source(
                    url=url,
                    name=item["name"],
                    domain=item["domain"],
                    country=item["country"],
                    language=item["language"],
                    region=item.get("region"),
                    priority=item.get("priority", "medium"),
                    tier=item.get("tier", 1),
                    status="active",
                    next_crawl_at=datetime.now(timezone.utc),  # 立即开始抓取
                )
                session.add(source)
                inserted += 1

    await engine.dispose()
    print(f"✅ Import complete: {inserted} inserted, {updated} updated, {skipped} skipped")
    print(f"   Total sources in YAML: {len(sources_data)}")


if __name__ == "__main__":
    asyncio.run(seed_sources())
```
